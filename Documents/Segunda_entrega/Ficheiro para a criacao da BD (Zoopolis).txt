-- Schema padrão
CREATE SCHEMA IF NOT EXISTS zoopolis AUTHORIZATION postgres;

-- Trocar para o schema
SET search_path TO zoopolis;

-- docker exec -it postgres-primary psql -U postgres -d postgres

/* Tabela para cadastro de animais */
CREATE TABLE animal (
                        ani_id SERIAL PRIMARY KEY,               -- Identificador único do animal
                        ani_name VARCHAR(60) NOT NULL,           -- Nome do animal
                        ani_ci_name VARCHAR(60),                 -- Nome científico do animal
                        ani_description TEXT,                    -- Descrição do animal
                        ani_weight FLOAT,                        -- Peso do animal
                        ani_height FLOAT,                        -- Altura do animal
                        ani_length FLOAT,                        -- Comprimento do animal
                        ani_cla_id INT,                          -- ID da classe associada ao animal
                        imageurl VARCHAR(255)                    -- URL da imagem do animal
);

/* Tabela para classes de animais */
CREATE TABLE class (
                       cla_id SERIAL PRIMARY KEY,               -- Identificador único da classe
                       cla_name VARCHAR(60) NOT NULL,           -- Nome da classe
                       cla_order VARCHAR(30),                   -- Ordem da classe
                       cla_fam VARCHAR(30)                      -- Família da classe
);

/* Tabela para registros de animais em enclosures */
CREATE TABLE ae (
                    ae_id SERIAL PRIMARY KEY,                -- Identificador único do registro
                    ae_ani_id INT,                           -- ID do animal
                    ae_enc_id INT,                           -- ID do enclosure
                    ae_dt_in TIMESTAMP,                      -- Data/hora de entrada do animal
                    ae_dt_out TIMESTAMP,                     -- Data/hora de saída do animal
                    ae_code VARCHAR(30)                      -- Código do animal
);

/* Tabela para detalhes de enclosures */
CREATE TABLE enclosure (
                           enc_id SERIAL PRIMARY KEY,               -- Identificador único do enclosure
                           enc_name VARCHAR(60) NOT NULL,           -- Nome do enclosure
                           enc_sup_amount INT,                      -- Quantidade de animais suportados
                           enc_aniclass VARCHAR(30),                -- Classe de animais suportada
                           enc_lat DOUBLE PRECISION NOT NULL,       -- Latitude do enclosure
                           enc_long DOUBLE PRECISION NOT NULL,      -- Longitude do enclosure
                           enc_sa_id INT                            -- ID da sub-área associada
);

/* Tabela para sub-áreas dentro de uma área */
CREATE TABLE sub_area (
                          sa_id SERIAL PRIMARY KEY,                -- Identificador único da sub-área
                          sa_area_id INT,                          -- ID da área associada
                          sa_name VARCHAR(30)                      -- Nome da sub-área
);

/* Tabela para áreas principais do zoológico */
CREATE TABLE area (
                      area_id SERIAL PRIMARY KEY,              -- Identificador único da área
                      area_name VARCHAR(30)                    -- Nome da área
);

/* Tabela para atividades realizadas no zoológico */
CREATE TABLE activity (
                          ac_id SERIAL PRIMARY KEY,                -- Identificador único da atividade
                          ac_name VARCHAR(50),                     -- Nome da atividade
                          ac_schedule TIMESTAMP,                   -- Horário agendado da atividade
                          ac_cap INT,                              -- Capacidade de espectadores
                          ac_area_id INT                           -- ID da área onde ocorre
);

/* Tabela para registro de visitas de pessoas */
CREATE TABLE visited (
                         vi_id SERIAL PRIMARY KEY,                -- Identificador único da visita
                         vi_per_id INT,                           -- ID da pessoa
                         vi_sa_id INT,                            -- ID da sub-área visitada
                         vi_dtime TIMESTAMP NOT NULL              -- Data/hora da visita
);

/* Tabela para cadastro de pessoas */
CREATE TABLE person (
                        per_id SERIAL PRIMARY KEY,               -- Identificador único da pessoa
                        per_name VARCHAR(60) NOT NULL,           -- Nome
                        per_email VARCHAR(30) UNIQUE,            -- Email
                        per_password VARCHAR(255) NOT NULL,      -- Senha
                        per_gender CHAR(1) NOT NULL,             -- Gênero
                        per_points INT NOT NULL                  -- Pontos
);

/* Tabela para registro de animais favoritos */
CREATE TABLE favorite (
                          fav_id SERIAL PRIMARY KEY,               -- Identificador único do favorito
                          fav_animal BOOLEAN,                      -- Indica se é um animal favorito
                          fav_ani_id INT,                          -- ID do animal
                          fav_per_id INT                           -- ID da pessoa
);

/* Tabela para quiosques */
CREATE TABLE kiosk (
                       kio_id SERIAL PRIMARY KEY,               -- Identificador único do quiosque
                       kio_name VARCHAR(30),                    -- Nome do quiosque
                       kio_area_id INT                          -- ID da área associada
);

/* Tabela para estoques de quiosques */
CREATE TABLE stock (
                       stock_id SERIAL PRIMARY KEY,             -- Identificador único do estoque
                       stock_amount INT,                        -- Quantidade em estoque
                       stock_kio_id INT,                        -- ID do quiosque
                       stock_pro_id INT                         -- ID do produto
);

/* Tabela para produtos disponíveis */
CREATE TABLE product (
                         pro_id SERIAL PRIMARY KEY,               -- Identificador único do produto
                         pro_name VARCHAR(100),                   -- Nome do produto
                         pro_barcode BIGINT UNIQUE CHECK (pro_barcode > 0),  -- Código de barras
                         pro_price DECIMAL(10,2)                  -- Preço
);

ALTER TABLE animal
    ADD CONSTRAINT animal_fk_class
        FOREIGN KEY (ani_cla_id) REFERENCES class(cla_id)

        FOREIGN KEY (ae_ani_id) REFERENCES animal(ani_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

    ADD CONSTRAINT ae_fk_enclosure
        FOREIGN KEY (ae_enc_id) REFERENCES enclosure(enc_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE enclosure

        FOREIGN KEY (enc_sa_id) REFERENCES sub_area(sa_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE sub_area
    ADD CONSTRAINT sub_area_fk_area
        FOREIGN KEY (sa_area_id) REFERENCES area(area_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE activity
    ADD CONSTRAINT activity_fk_area
        FOREIGN KEY (ac_area_id) REFERENCES area(area_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE visited
    ADD CONSTRAINT visited_fk_person
        FOREIGN KEY (vi_per_id) REFERENCES person(per_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE visited
    ADD CONSTRAINT visited_fk_sub_area
        FOREIGN KEY (vi_sa_id) REFERENCES sub_area(sa_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE favorite
    ADD CONSTRAINT favorite_fk_animal
        FOREIGN KEY (fav_ani_id) REFERENCES animal(ani_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE favorite
    ADD CONSTRAINT favorite_fk_person
        FOREIGN KEY (fav_per_id) REFERENCES person(per_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE kiosk
    ADD CONSTRAINT kiosk_fk_area
        FOREIGN KEY (kio_area_id) REFERENCES area(area_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE stock
    ADD CONSTRAINT stock_fk_kiosk
        FOREIGN KEY (stock_kio_id) REFERENCES kiosk(kio_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE stock
    ADD CONSTRAINT stock_fk_product
        FOREIGN KEY (stock_pro_id) REFERENCES product(pro_id)
            ON DELETE NO ACTION ON UPDATE NO ACTION;

ALTER TABLE zoopolis.favorite
    ALTER COLUMN fav_animal TYPE VARCHAR(255)
        USING fav_animal::text;

    ALTER COLUMN per_gender TYPE VARCHAR(255)
        USING per_gender::text;

-- Criar usuário de replicação se não existir
DO $$
    BEGIN
        IF NOT EXISTS (SELECT FROM pg_catalog.pg_roles WHERE rolname = 'replicator') THEN
            CREATE ROLE replicator WITH REPLICATION LOGIN PASSWORD 'ThZ3d1112';
        END IF;
    END
$$;

-- Conceder permissões
GRANT CONNECT ON DATABASE postgres TO replicator;
GRANT USAGE ON SCHEMA zoopolis TO replicator;
GRANT SELECT ON ALL TABLES IN SCHEMA zoopolis TO replicator;

-- Criar replication slot
SELECT pg_create_physical_replication_slot('replica_slot', true);
